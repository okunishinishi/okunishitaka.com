#!/usr/bin/env node

/**
 * Prerender web page with phantomjs.
 */

"use strict";


var args = require('system').args,
    url = args[0],
    filename = args[1];

var page = require('webpage').create(),
    fs = require('fs');

console.log('args', args);
console.log('url', url);
page.open(url, function () {
    fs.write(filename, page.content, function (err) {
        if (err) {
            console.error(err);
        }
        phantom.exit();
    });
});
